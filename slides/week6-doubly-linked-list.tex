\begin{frame}

	\scalebox{.85}{
		\begin{algorithm}[H]
			\setstretch{1}
			\caption{removeLast(head, tail)}
			\If{isEmpty(head)} {
				\tcp{case 1: it is empty}
				raise an error
			}\ElseIf{head == tail} {
				\tcp{case 2: the size is 1}
				$head\gets null$ \\
				$tail\gets null$
			}\Else{
				\tcp{case 3: the size is greater than 1}
				$x\gets tail.prev$\\
				$x.next\gets null$ \\
				$tail\gets x$
			}
		\end{algorithm}
	}

\end{frame}

			\caption{sortedInsert(head, k)}
			$node\gets Node(k)$ \\

				$x.next\gets node$
			}

	\scalebox{.85}{
		\begin{algorithm}[H]
			\caption{reverseList(head)}
			$pre\gets null$ \\
			$cur\gets head$ \\
			\While{cur $\neq$ null} {
				$next\gets cur.next$ \\
				$cur.next\gets pre$ \\
				$pre\gets cur$ \\
				$cur\gets next$
			}
			\Return{pre}
		\end{algorithm}
	}
